// ── MARKET PULSE STRIP ──────────────────────────────────────────────────────── const _pulseAPI = window.location.hostname==='localhost'||window.location.hostname==='127.0.0.1' ? 'http://localhost:8000' : 'https://oanda-trading-center.onrender.com'; let _pulsePrev = {}; function fmtPulse(v, decimals) { return Number(v).toFixed(decimals); } function setPulseItem(valId, chgId, price, prevPrice, decimals) { const valEl = document.getElementById(valId); const chgEl = document.getElementById(chgId); if (!valEl || !chgEl) return; valEl.textContent = fmtPulse(price, decimals); if (prevPrice && prevPrice > 0) { const chgPct = ((price - prevPrice) / prevPrice * 100); const arrow = chgPct > 0 ? '▲' : chgPct < 0 ? '▼' : '—'; const cls = chgPct > 0 ? 'pulse-up' : chgPct < 0 ? 'pulse-dn' : 'pulse-neu'; valEl.className = `pulse-val ${cls}`; chgEl.className = `pulse-chg ${cls}`; chgEl.textContent = `${arrow} ${Math.abs(chgPct).toFixed(2)}%`; } else { valEl.className = 'pulse-val pulse-neu'; chgEl.textContent = ''; } } function updateSessions() { const h = new Date().getUTCHours(); const m = new Date().getUTCMinutes(); const t = h + m/60; const sessions = { 'sess-lon': t >= 7 && t < 17, 'sess-ny': t >= 13 && t < 22, 'sess-tok': t >= 0 && t < 9, 'sess-syd': t >= 21 || t < 7, }; Object.entries(sessions).forEach(([id, isOpen]) => { const el = document.getElementById(id); if (!el) return; el.className = `pulse-sess-item ${isOpen ? 'on' : 'off'}`; el.querySelector('.pulse-dot').className = `pulse-dot ${isOpen ? 'on' : 'off'}`; }); } async function updatePulse() { try { const r = await fetch(`${_pulseAPI}/api/prices`); const d = await r.json(); if (!d.ok) return; const p = d.prices; // DXY proxy = inverse of EUR/USD // When EUR/USD rises, DXY falls and vice versa if (p.EUR_USD) { const eurusd = p.EUR_USD.mid; const dxyProxy = 1 / eurusd * 100; // normalised proxy const prevEur = _pulsePrev.EUR_USD; const prevDxy = prevEur ? 1/prevEur*100 : null; // DXY is INVERSE of EUR/USD direction setPulseItem('p-dxy','p-dxy-c', dxyProxy, prevDxy, 2); // Flip the colour for DXY (EUR up = DXY down) const dxyEl = document.getElementById('p-dxy'); if (dxyEl && prevDxy) { const isUp = dxyProxy > prevDxy; dxyEl.className = `pulse-val ${isUp?'pulse-up':'pulse-dn'}`; const chgEl = document.getElementById('p-dxy-c'); if(chgEl){ const pct = Math.abs((dxyProxy-prevDxy)/prevDxy*100); chgEl.className = `pulse-chg ${isUp?'pulse-up':'pulse-dn'}`; chgEl.textContent = `${isUp?'▲':'▼'} ${pct.toFixed(2)}%`; } } _pulsePrev.EUR_USD = eurusd; } if (p.XAU_USD) { setPulseItem('p-xau','p-xau-c', p.XAU_USD.mid, _pulsePrev.XAU_USD, 2); _pulsePrev.XAU_USD = p.XAU_USD.mid; } if (p.WTICO_USD) { setPulseItem('p-wti','p-wti-c', p.WTICO_USD.mid, _pulsePrev.WTICO_USD, 2); _pulsePrev.WTICO_USD = p.WTICO_USD.mid; } if (p.SPX500_USD) { setPulseItem('p-spx','p-spx-c', p.SPX500_USD.mid, _pulsePrev.SPX500_USD, 1); _pulsePrev.SPX500_USD = p.SPX500_USD.mid; } if (p.GBP_JPY) { setPulseItem('p-gbp','p-gbp-c', p.GBP_JPY.mid, _pulsePrev.GBP_JPY, 3); _pulsePrev.GBP_JPY = p.GBP_JPY.mid; } } catch(e) {} updateSessions(); } // Start pulse updatePulse(); setInterval(updatePulse, 30000); setInterval(updateSessions, 60000);
